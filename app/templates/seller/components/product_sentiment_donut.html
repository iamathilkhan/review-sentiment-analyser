{% macro sentiment_donut(positive, negative, neutral, size=60) %} {% set total = positive + negative + neutral %} {% if total == 0 %} {% set pos_per, neg_per, neu_per = 0, 0, 100 %} {% else %} {% set pos_per = (positive / total * 100) %} {% set neg_per = (negative / total * 100) %} {% set neu_per = (neutral / total * 100) %} {% endif %} {% set radius = (size / 2) - 4 %} {% set circumference = 2 * 3.14159 * radius %} {% set pos_stroke = (pos_per / 100) * circumference %} {% set neg_stroke = (neg_per / 100) * circumference %} {% set neu_stroke = (neu_per / 100) * circumference %}
{{ (pos_per)|round|int }}%
{% endmacro %}
